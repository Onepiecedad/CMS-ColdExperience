# frontend/netlify.toml
[build]
  command = "npm run build"
  publish = "build"

[functions]
  directory = "netlify/functions"

[build.processing]
  skip_processing = false

[build.processing.css]
  bundle = true
  minify = true

[build.processing.js]
  bundle = true
  minify = true

[build.processing.html]
  pretty_urls = true

[build.processing.images]
  compress = true

[dev]
  command = "npm start"
  port = 3000

[build.environment]
  REACT_APP_API_URL = "https://coldexperience.onrender.com"
  REACT_APP_SITE_URL = "https://coldexperience.se"

[[redirects]]
  from = "/*"
  to = "/index.html"
  status = 200

  from = "/en/packages"

  status = 301

  to = "/sv/paketresor"
  status = 301

[[redirects]]

  to = "/de/lappland-reisepakete"
  status = 301

  [[headers]]
    for = "/*"
    [headers.values]
      X-Robots-Tag = "noindex, nofollow"

  [[headers]]
    for = "/*"
